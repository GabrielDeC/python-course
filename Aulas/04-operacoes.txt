# Soma --> +
# Subtração --> -
# Multiplicação --> *
# Divisão --> /
# Potência --> **
# Divisão Inteira --> //
# Resto da divisão (Módulo) --> %
# Sinal de atribuição ou "Recebe" --> ==
# Ordem de Precedência --> () >> ** >> / , * , // , % >> + e -

a = 5 % 2

print("O resultado é {:.3f}".format(a))

print("O resultado foi {:.3f}.".format(a), end=' >>> ')
print("Teste de não \n quebra de linha")
